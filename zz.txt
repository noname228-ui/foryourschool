        function toggleRoomPanel() {
            const panel = document.getElementById('room-panel');

        }

        function send() {
            let u = document.getElementById('username').value;
            let t = document.getElementById('text').value;
            if(!t.trim()) return;
            let path = currentRoom === "global" ? "global_chat" : "rooms/" + currentRoom + "/messages";
            db.ref(path).push({ u: u, t: t.replace(badWords, "***") });
            document.getElementById('text').value = "";
        }

        function load() {
            let path = currentRoom === "global" ? "global_chat" : "rooms/" + currentRoom + "/messages";
            db.ref(path).off();
            db.ref(path).limitToLast(50).on('value', (snapshot) => {
                let html = "";
                snapshot.forEach(child => {
                    let d = child.val();
                    if(d.u && d.t) html += `<div class="msg"><b>${d.u}</b>${d.t}</div>`;
                });
                const chatBox = document.getElementById('chat');
                chatBox.innerHTML = html;
                chatBox.scrollTop = chatBox.scrollHeight;
            });
        }

        function joinRoom() {
            let r = document.getElementById('roomName').value.trim();
            let p = document.getElementById('roomPass').value.trim();
            if(!r || !p) { alert("Enter Name and Pass!"); return; }
            
            db.ref('rooms/' + r + '/pass').once('value', (s) => {
                if(s.exists() && s.val() !== p) {
                    alert("Wrong password!");
                } else {
                    if(!s.exists()) db.ref('rooms/' + r + '/pass').set(p);
                    currentRoom = r;
                    document.getElementById('status').innerText = "Current Room: " + r;
                    document.getElementById('room-panel').style.display = 'none';
                    load();
                }
            });
        }

        function goGlobal() {
            currentRoom = "global";
            document.getElementById('status').innerText = "Current Room: Global";
            load();
        }
